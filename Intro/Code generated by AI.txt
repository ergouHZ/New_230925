（These are used for selected different content by clicking the button on the navbar, since I don't want to do it in the way of scrolling, I search these codes for using, and I think it's really efficient）

fetch(url)
    .then(response => response.text())
    .then(content => {
        // Replace the content of the "container" div with the loaded content
        document.getElementById('container').innerHTML = content;
    })
    .catch(error => {
        console.error('Error loading page: ' + error);
    });
